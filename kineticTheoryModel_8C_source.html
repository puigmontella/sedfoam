322void Foam::kineticTheoryModel::solve
323(
324 const volTensorField& gradUat,
325 const volScalarField& kb,
326 const volScalarField& epsilonb,
327 const volScalarField& nutf,
328 const dimensionedScalar& B,
329 const dimensionedScalar& tt
330)
331{
332 if (kineticTheory_ == false)
333 {
334 return;
335 }
337 //Define some usefull quantities
339 const scalar sqrtPi(sqrt(constant::mathematical::pi));
340 dimensionedScalar alphaSmall
341 (
342 "small",
343 dimensionSet(0, 0, 0, 0, 0, 0, 0),
344 scalar(1.0e-4)
345 );
346 dimensionedScalar Tsmall
347 (
348 "small",
349 dimensionSet(0, 2, -2, 0, 0, 0, 0),
350 SMALL
351 );
352 volScalarField ThetaSqrt(sqrt(Theta_));
353 dimensionedScalar Tpsmall_
354 (
355 "Tpsmall_",
356 dimensionSet(1, -1, -3, 0, 0, 0, 0),
357 scalar(1e-30)
358 );
359 volTensorField dU(fvc::grad(Ua_));//gradUat.T(); //that is fvc::grad(Ua_);
360 volSymmTensorField D(symm(dU)); //0.5*(dU + dU.T)
361
362 volScalarField ThetaClip(Theta_);
363 if (limitProduction_) // limit the production by clipping Theta to 2/3kb
364 {
365 ThetaClip = min(Theta_, (2.0/3.0)*max(kb));
366 }
367
369 // Radial distribution function g0
371 gs0_ = radialModel_->g0(min(alpha_, alphaMax_ - alphaSmall), alphaMax_);
372
374 // collisional pressure (Eq. 3.22, p. 45)
375 // ///////////////////////////////////////
376 volScalarField PsCoeff
377 (
378 granularPressureModel_->granularPressureCoeff
379 (
380 alpha_,
381 gs0_,
382 rhoa_,
383 e_
384 )
385 );
386
388 // Temperature conductivity (Table 3.3, p. 49)
390 kappa_ = conductivityModel_->kappa(alpha_, ThetaClip, gs0_, rhoa_, da_, e_);
391
393 // Granular viscosity (Table 3.2, p.47)
394 // and shear stress
396 mua_ = viscosityModel_->mua(alpha_, ThetaClip, gs0_, rhoa_, da_, e_);
397 // bulk viscosity p. 45.
398 lambda_ = viscosityModel_->lambda(alpha_, ThetaClip, gs0_, rhoa_, da_, e_);
399
400 // stress tensor, Definitions, Table 3.1, p. 43
401 volSymmTensorField tau
402 (
403 2.0*mua_*D + (lambda_ - (2.0/3.0)*mua_)*tr(D)*I
404 );
406 // Contact dissipation
408
409 // Correlation length for the extended kinetic theory
410 volScalarField Lc
411 (
412 da_*(alpha_+alphaSmall)/(alpha_+alphaSmall)
413 );
414 if (extended_)
415 {// extended kinetic theory Jenkins (2007)
416 Lc = da_*max
417 (
418 scalar(1),
419 0.5*pow(30./(1.+sqr(sqrtPi)/12.)*(1-e_)*sqr(alpha_)*gs0_, 1./3.)
420 );
421 }
422 // Inelastic dissipation (Eq. 3.24, p.50)
423 volScalarField gammaCoeff
424 (
425 (3.0*(1.0 - sqr(e_))*sqr(alpha_)*rhoa_*gs0_
426 *(4.0/Lc*ThetaSqrt/sqrtPi-tr(D)))
427 );
428 // Frictional dissipation (Chialvo and Sundaresan (2013) eqs. 22-23)
429 dimensionedScalar f_mu(3./2*muPart_*exp(-3*muPart_));
430 dimensionedScalar e_eff(e_ - f_mu);
431 dimensionedScalar fric_correction((1-pow(e_eff, 2))/(1-pow(e_, 2)));
432 gammaCoeff *= fric_correction;
433
435 // Work of the drag force by granular fluctuating velocity
436 // NB, drag = K*alpha*beta,
437 // this is inconsistent with momentum equation,
438 // since the following form missed the drift velocity
440 volScalarField K(draga_.K(mag(Ua_ - Ub_)));
441 // The following is to calculate parameter tmf_ in u_f*u_s correlation
442 // (Danon et al. (1977))
443 volScalarField flucVelCor_
444 (
445 Foam::exp(-B*rhoa_*6*epsilonb/max(kb*(1-alpha_)*K, Tpsmall_))
446 );
447 // Eq. 3.25, p. 50 J_int = J2 - J1*Theta_
448 volScalarField J1(alpha_*(1-alpha_)*K*(3+2*quadraticCorrectionJ1_));
449 volScalarField J2(alpha_*(1-alpha_)*K*(2*flucVelCor_*kb));
450
452 // Granular temperature equation
453 // /////////////////////////////
454 surfaceScalarField phi
455 (
456 1.5*rhoa_*phia_*fvc::interpolate((alpha_+alphaSmall))
457 );
458 // construct the granular temperature equation (Eq. 3.20, p. 44)
459 // NB. note that there are two typos in Eq. 3.20
460 // no grad in front of Ps
461 // wrong sign in front of laplacian
462 fvScalarMatrix ThetaEqn
463 (
464 fvm::ddt(1.5*(alpha_+alphaSmall)*rhoa_, Theta_)
465 + fvm::div(phi, Theta_, "div(phi,Theta)")
466 ==
467 // Ps term.
468 fvm::SuSp(-PsCoeff*fvc::div(phia_), Theta_)
469 // production due to shear.
470 + (tau && dU)
471 // granular temperature conduction.
472 + fvm::laplacian(kappa_, Theta_, "laplacian(kappa,Theta)")
473 // energy disipation due to inelastic collision.
474 + fvm::Sp(-gammaCoeff, Theta_)
475 // dissipation through drag force
476 + fvm::Sp(-J1, Theta_)
477 // transfer of fluctuating kinetic energy from fluid to particles
478 + (1-killJ2_)*J2
479 );
480
481 if (writeTBudget_)
482 {
483 #include "writeTBudget.H"
484 }
485 ThetaEqn.relax();
486 ThetaEqn.solve();
487
488 // Limit the value of temperature between Tsmall and MaxTheta
489 Theta_.max(Tsmall);
490 Theta_.min(MaxTheta);
491
492 // Update coefficients after solving Theta
493 PsCoeff = granularPressureModel_->granularPressureCoeff(
494 alpha_, gs0_, rhoa_, e_);
495 pa_ = PsCoeff*Theta_;
496 mua_ = viscosityModel_->mua(alpha_, Theta_, gs0_, rhoa_, da_, e_);
497 lambda_ = viscosityModel_->lambda(alpha_, Theta_, gs0_, rhoa_, da_, e_);
498
499 pa_.max(0.0);
500 pa_.correctBoundaryConditions();
501
502 mua_.max(0.0);
503 mua_.correctBoundaryConditions();
504
505 Info << "Granular temp. Theta: Min ="<< gMin(Theta_)
506 << " Max = "<< gMax(Theta_)<< endl;
507}
