53Foam::tmp<Foam::volScalarField> Foam::granularRheologyModels::MuIdimLess::muI
54(
55 const dimensionedScalar& mus,
56 const dimensionedScalar& mu2,
57 const dimensionedScalar& I0,
58 const volScalarField& pa,
59 const dimensionedScalar& rhoa,
60 const dimensionedScalar& da,
61 const dimensionedScalar& rhob,
62 const dimensionedScalar& nub,
63 const volScalarField& magD,
64 const dimensionedScalar& Dsmall
65) const
66{
67 // MuI inertial regime friction model: mu(I) = mus + (mu2-mus)/(I0/I+1)
68 // with I=da magD / sqrt(pa)
69 return mus + (mu2 - mus)*magD/(I0*pow(pa, 0.5)/da + magD + Dsmall);
70}

72Foam::tmp<Foam::volScalarField> Foam::granularRheologyModels::MuIdimLess::I
73(
74 const volScalarField& pa,
75 const dimensionedScalar& rhoa,
76 const dimensionedScalar& da,
77 const dimensionedScalar& rhob,
78 const dimensionedScalar& nub,
79 const volScalarField& magD
80) const
81{
82 // inertial regime I=da magD / sqrt(pa)
83 return magD * da * pow(pa, -0.5);
84}
