36template<class BasicTurbulenceModel>
37void twophasekEpsilon<BasicTurbulenceModel>::correctNut()
38{
39 this->nut_ = Cmu_*sqr(k_)/epsilon_;
40 this->nut_.min(nutMax_);
41 this->nut_.correctBoundaryConditions();
42 fv::options::New(this->mesh_).correct(this->nut_);
43
44 BasicTurbulenceModel::correctNut();
45}

233template<class BasicTurbulenceModel>
234void twophasekEpsilon<BasicTurbulenceModel>::correct()
235{
236 if (not this->turbulence_)
237 {
238 return;
239 }
240
241 // Local references
242 const volVectorField& U = this->U_;
243 volScalarField& nut = this->nut_;
244 const surfaceScalarField& phi = this->phi_;
245 fv::options& fvOptions(fv::options::New(this->mesh_));
246
247 eddyViscosity<RASModel<BasicTurbulenceModel>>::correct();
248
249 volScalarField divU(fvc::div(fvc::absolute(this->phi(), U)));
250
251 volTensorField GradU(fvc::grad(U));
252 volSymmTensorField Sij(symm(GradU));
253
254 //volScalarField::Internal G
255 volScalarField G
256 (
257 this->GName(),
258 nut*(GradU && dev(twoSymm(GradU)))
259 );
260
261 // Update omega and G at the wall
262 epsilon_.boundaryFieldRef().updateCoeffs();
263
264 // Dissipation equation
265 tmp<fvScalarMatrix> epsEqn
266 (
267 fvm::ddt(epsilon_)
268 + fvm::div(phi, epsilon_)
269 - fvm::Sp(fvc::div(phi), epsilon_)
270 - fvm::laplacian(DepsilonEff(), epsilon_)
271 ==
272 - fvm::SuSp(-C1_*G/k_, epsilon_)
273 + fvm::Sp(-C2_*epsilon_/k_, epsilon_)
274 + fvm::Sp(C3ep_*ESD_, epsilon_)
275 + ESD2()*fvm::Sp(C3ep_*KE2_, epsilon_)
276 + fvm::Sp(C4ep_*KE4_*ESD5_*nut/k_, epsilon_)
277 );
278 if (writeTke_)
279 {
280 #include "writeTKEBudget_kEpsilon.H"
281 }
282 epsEqn.ref().relax();
283 fvOptions.constrain(epsEqn.ref());
284 epsEqn.ref().boundaryManipulate(epsilon_.boundaryFieldRef());
285 solve(epsEqn);
286 fvOptions.correct(epsilon_);
287 bound(epsilon_, this->epsilonMin_);
288
289
290 // Turbulent kinetic energy equation
291 tmp<fvScalarMatrix> kEqn
292 (
293 fvm::ddt(k_)
294 + fvm::div(phi, k_)
295 - fvm::Sp(fvc::div(phi), k_)
296 - fvm::laplacian(DkEff(), k_)
297 ==
298 - fvm::SuSp(-G/k_, k_)
299 - fvm::Sp(epsilon_/k_, k_)
300 + fvm::Sp(ESD_, k_)
301 + fvm::Sp(KE4_*ESD4_*nut/k_, k_)
302 + ESD2()*fvm::Sp(KE2_, k_)
303 );
304
305 kEqn.ref().relax();
306 fvOptions.constrain(kEqn.ref());
307 solve(kEqn);
308 fvOptions.correct(k_);
309 bound(k_, this->kMin_);
310
311 correctNut();
312}
