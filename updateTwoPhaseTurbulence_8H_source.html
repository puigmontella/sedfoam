21// Compute two-phase specific terms for RAS turbulence models:
22// drag dissipation and stratification terms.

24// Exponent of the turbulence damping term for k,epsilon and omega equations
25tmfexp = -B*rhoa*6.0/(max(beta*K, Tpsmall));

27// just need the horizontal ones;
28rtemp = (Ub-Ua) & gravity;
29Urtemp = Ub-Ua;
30if (Foam::mag(gravity).value() != 0)
31{
32 Urtemp = rtemp*gravity/magSqr(gravity);
33}
34Urh = (Ub-Ua) - Urtemp;
35
36if (not SUSaniso)
37{
38// ESD1 for horizontal, ESD4 for vertical density stratification
39// for 1D ESD1 ==0, obviously
40ESD1 = (K/rhob)*(fvc::grad(alpha)) & Urh;
41
42// stratification
43ESD4 = -(SUS/beta)*(rhoa/rhob-1.0)*gravity & fvc::grad(alpha);
44ESD4 = ESD4 + KE1*ESD1;
45
46// the following term is implemented in epsilon equation.
47ESD5 = -(SUS/beta)*(rhoa/rhob-1.0)*gravity & fvc::grad(alpha);
48}
49else
50{
51// ESD1 for horizontal, ESD4 for vertical density stratification
52// for 1D ESD1 ==0, obviously
53ESD1 = SUS*(K/rhob)*(SUS_I*iso-SUS_A*aniso)&(fvc::grad(alpha))
54 & Urh;
55
56// stratification
57ESD4 = - SUS/beta*(rhoa/rhob-1.0)*gravity
58 & fvc::grad(alpha);
59ESD4 = ESD4 + KE1*ESD1;
60
61// the following term is implemented in epsilon equation.
62ESD5 = - SUS/beta*(rhoa/rhob-1.0)*gravity
63 & fvc::grad(alpha);
64}
65// if ESD5 > 0, then it should be implemented, else, no
66ESD5 = max(ESD5, scalar(0.0)*ESD5);
67ESD5 = ESD5 + KE1*ESD1;
68
69// k was included in the equation implicitly
70ESD3 = -2.0*K*alpha/rhob;
71
72ESD = KE3*ESD3; //need to check here!
