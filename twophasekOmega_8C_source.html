264template<class BasicTurbulenceModel>
265void twophasekOmega<BasicTurbulenceModel>::correct()
266{
267 if (not this->turbulence_)
268 {
269 return;
270 }
271
272 // Local references
273 const alphaField& beta = this->alpha_;
274 const alphaField alpha = 1 - beta;
275 const volVectorField& U = this->U_;
276 volScalarField& nut = this->nut_;
277 const surfaceScalarField& phi = this->phi_;
278 fv::options& fvOptions(fv::options::New(this->mesh_));
279
280 eddyViscosity<RASModel<BasicTurbulenceModel>>::correct();
281
282 volScalarField divU(fvc::div(fvc::absolute(this->phi(), U)));
283
284 volSymmTensorField Sij(symm(fvc::grad(U)));
285
286 volScalarField G
287 (
288 this->GName(),
289 nut*2*magSqr(Sij)
290 );
291
292 // Update omega and G at the wall
293 omega_.boundaryFieldRef().updateCoeffs();
294
295 volTensorField Omij(-skew(fvc::grad(U)));
296 volVectorField Gradk(fvc::grad(k_));
297 volVectorField Gradomega(fvc::grad(omega_));
298 volScalarField alphadCheck_(Gradk & Gradomega);
299
300 const volScalarField CDkOmega
301 (
302 sigmad_*pos(0.15-alpha)*pos(alphadCheck_)*(alphadCheck_)/omega_
303 );
304
305
306 volScalarField XsiOmega
307 (
308 IOobject
309 (
310 IOobject::groupName("XsiOmega", U.group()),
311 this->runTime_.timeName(),
312 this->mesh_,
313 IOobject::NO_READ,
314 IOobject::NO_WRITE
315 ),
316 this->mesh_,
317 dimensionedScalar("zero", dimless, 0.0)
318 );
319 if (popeCorrection_)
320 {
321 XsiOmega =
322 (
323 mag((Omij & Omij & Sij)/(pow((Cmu_*omega_), 3)))
324 );
325 }
326
327 // Turbulence specific dissipation rate equation
328 tmp<fvScalarMatrix> omegaEqn
329 (
330 fvm::ddt(omega_)
331 + fvm::div(phi, omega_)
332 - fvm::Sp(fvc::div(phi), omega_)
333 - fvm::laplacian(DomegaEff(), omega_)
334 ==
335 - fvm::SuSp (-alphaOmega_*G/k_, omega_)
336 - fvm::Sp(ESD_, omega_)
337 - fvm::Sp
338 (
339 betaOmega_*
340 (
341 (scalar(1.0)+scalar(85.0)*XsiOmega())
342 /(scalar(1.0)+scalar(100.0)*XsiOmega())
343 )*omega_(),
344 omega_
345 )
346 + CDkOmega
347 + ESD2()*fvm::Sp(C3om_*KE2_, omega_)
348 + fvm::Sp((C4om_*KE4_*ESD5_*nut/k_), omega_)
349 );
350 if (writeTke_)
351 {
352 #include "writeTKEBudget_kOmega.H"
353 }
354 omegaEqn.ref().relax();
355 fvOptions.constrain(omegaEqn.ref());
356 omegaEqn.ref().boundaryManipulate(omega_.boundaryFieldRef());
357 solve(omegaEqn);
358 fvOptions.correct(omega_);
359 bound(omega_, this->omegaMin_);
360
361
362 // Turbulent kinetic energy equation
363 tmp<fvScalarMatrix> kEqn
364 (
365 fvm::ddt(k_)
366 + fvm::div(phi, k_)
367 - fvm::Sp(fvc::div(phi), k_)
368 - fvm::laplacian(DkEff(), k_)
369 ==
370 - fvm::SuSp(-G/k_, k_)
371 + fvm::Sp(-Cmu_*omega_, k_)
372 + fvm::Sp(ESD_, k_)
373 + fvm::Sp(KE4_*ESD4_*nut/k_, k_)
374 + ESD2()*fvm::Sp(KE2_, k_)
375 );
376
377 kEqn.ref().relax();
378 fvOptions.constrain(kEqn.ref());
379 solve(kEqn);
380 fvOptions.correct(k_);
381 bound(k_, this->kMin_);
382
383 correctNut();
384}
