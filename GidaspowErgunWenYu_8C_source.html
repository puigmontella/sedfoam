65Foam::tmp<Foam::volScalarField> Foam::GidaspowErgunWenYu::K
66(
67 const volScalarField& Ur
68) const
69{
70 volScalarField beta
71 (
72 max(scalar(1) - alpha_, scalar(1.0e-6))
73 );
74
75 // volScalarField bp = pow(beta, -2.65);
76 volScalarField bp(pow(beta, -phasea_.hExp()));
77 volScalarField Re
78 (
79 max(beta*Ur*phasea_.d()*phasea_.sF()/phaseb_.nu(),
80 scalar(1.0e-9))
81 );
82
83 volScalarField Cds
84 (
85 neg(Re - 1000)*(24.0*(1.0 + 0.15*pow(Re, 0.687))/Re)
86 + pos(Re - 1000)*0.44
87 );
88
89/*
90 // Wen and Yu (1966)
91 // modified in May 17, 2012, by C.Z.
92 //tmp<volScalarField> tKWenYu = (0.75*Cds*phaseb_.rho()*Ur*bp
93 /(phasea_.d()*phasea_.sF()));
94 volScalarField tKWenYu = (0.75*Cds*phaseb_.rho()*Ur*bp
95 /(phasea_.d()*phasea_.sF()));
96 volScalarField& KWenYu = tKWenYu();
97
98 // Ergun
99 forAll(beta, cellj)
100 {
101 if (beta[cellj] < 0.8)
102 {
103 tKWenYu[cellj] =
104 150.0*alpha_[cellj]*phaseb_.nu().value()*phaseb_.rho().value()
105 /sqr(beta[cellj]*phasea_.d().value())
106 + 1.75*phaseb_.rho().value()*Ur[cellj]
107 /(beta[cellj]*phasea_.d().value());
108 }
109 }
110// WARNING: remove this line will makes the parallel computations "instable"
111 tKWenYu.correctBoundaryConditions();
112
113 return KWenYu;
114*/
115// pos0(beta - 0.8)*(0.75*Cds*phaseb_.rho()*Ur*bp
116// /(phasea_.d()*phasea_.sF())) //line for sedfoam-5.0
117 return pos(beta - 0.8)
118 *(0.75*Cds*phaseb_.rho()*Ur*bp/(phasea_.d()*phasea_.sF()))
119 + neg(beta - 0.8)
120 *(150.0*alpha_*phaseb_.nu()*phaseb_.rho()/sqr(beta*phasea_.d())
121 + 1.75*phaseb_.rho()*Ur/(beta*phasea_.d()));
122}
